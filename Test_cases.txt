# Addition (positive even B)
10 2 5 = 15
10 4 5 = 15
-10 6 5 = -5
12 100 8 = 20

# Subtraction (negative even B)
10 -2 5 = 5
10 -4 5 = 5
-10 -6 5 = -15
12 -100 8 = 4

# Multiplication (positive odd B)
10 1 5 = 50
10 3 5 = 50
-10 5 5 = -50
12 101 8 = 96

# Division (negative odd B)
10 -1 5 = 2
10 -3 5 = 2
-10 -5 5 = -2
24 -101 8 = 3

# Decimal Inputs
5.5 2 2.5 = 8.0
5.5 -2 2.5 = 3.0
5.5 1 2.5 = 13.75
5.5 -1 2.5 = 2.2

# Zero as A
0 2 5 = 5
0 -2 5 = -5
0 1 5 = 0
0 -1 5 = 0

# Zero as C
5 2 0 = 5
5 -2 0 = 5
5 1 0 = 0
5 -1 0 = ZeroDivisionError

# Large Numbers
1000000 2 999999 = 1999999
1000000 -2 999999 = 1
1000000 1 999999 = 999999000000
1000000 -1 100 = 10000

# Invalid Operation Code
5 0 2 = Invalid / Error
